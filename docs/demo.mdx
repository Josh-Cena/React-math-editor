# Demo

import { EditorProvider, Input, Output } from 'react-math-editor';
import Highlight, { defaultProps } from 'prism-react-renderer';
import usePrismTheme from '@theme/hooks/usePrismTheme';

## Basic example

```tsx
function Editor() {
  return (
    <EditorProvider>
      <Input />
      <Output />
    </EditorProvider>
  );
}
```
